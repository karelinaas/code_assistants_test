Привет!

У меня есть следующие таблицы для БД на движке PostgreSQL:

```sql
-- Таблица с пользователями
create table referrals_affiliate
(
    id           bigint generated by default as identity
        primary key,
    password     varchar(128)             not null,
    last_login   timestamp with time zone,
    is_superuser boolean                  not null,
    username     varchar(150)             not null
        unique,  -- имя пользователя
    first_name   varchar(150)             not null,
    last_name    varchar(150)             not null,
    is_staff     boolean                  not null,
    is_active    boolean                  not null,
    date_joined  timestamp with time zone not null,
    created_at   timestamp with time zone not null,
    updated_at   timestamp with time zone not null,
    email        varchar(254)             not null
        unique
);

-- Таблица с реферальными программами
create table referrals_campaign
(
    id             bigint generated by default as identity
        primary key,
    created_at     timestamp with time zone not null,
    updated_at     timestamp with time zone not null,
    is_deactivated boolean                  not null,
    title          varchar(255)             not null,
    description    varchar(255),
    promocode      varchar(20)              not null,  -- промокод программы
    reward         double precision         not null,
    active_since   date                     not null,
    active_till    date                     not null,
    discount       integer
);

create table referrals_referralstat
(
    id               bigint generated by default as identity
        primary key,
    created_at       timestamp with time zone not null,
    updated_at       timestamp with time zone not null,
    referrals_number integer                  not null,  -- кол-во рефералов
    affiliate_id     bigint                   not null
        constraint referrals_referralst_affiliate_id_a13f8035_fk_referrals
            references referrals_affiliate
            deferrable initially deferred,
    campaign_id      bigint                   not null
        constraint referrals_referralst_campaign_id_d0d0dd3f_fk_referrals
            references referrals_campaign
            deferrable initially deferred,
    constraint referrals_referralstat_campaign_id_affiliate_id_0a96f462_uniq
        unique (campaign_id, affiliate_id)
);
```

Напиши мне, пожалуйста, sql-запрос, показывающий сравнение кол-ва рефералов, приведенных в программу каждым пользователем, со средним кол-вом рефералов по программе.
Также включи в запрос поля с промокодом программы и именем пользователя.
